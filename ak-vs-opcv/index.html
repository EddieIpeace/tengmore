// Standard Library
#include <array>
#include <iostream>
#include <string>
#include <fstream>
#include <conio.h>
// OpenCV Header
#include <opencv2/core.hpp>
#include <opencv2/imgproc.hpp>
#include <opencv2/highgui.hpp>
#include "opencv2/opencv.hpp"  
#include<opencv2\highgui\highgui.hpp>
#include<opencv2\imgproc\imgproc.hpp>
#pragma comment(lib, "k4a.lib")
#include <stdio.h>
#include <opencv2\highgui\highgui_c.h>      
#include <vector>   
#include < iostream>
#include <stdlib.h>
// Kinect for Windows SDK Header
#include <k4a/k4a.h>
#include <k4arecord/record.h>
#include <k4arecord/playback.h>
#include <k4a/k4a.hpp>
#include < cstdlib>

using namespace std;
using namespace cv;

int main(int argc, char** argv)
{
	k4a::capture capture;
	const uint32_t device_count = k4a::device::get_installed_count();//查看设备个数
	if (device_count == 0)
	{
		printf("No K4A devices found\n");
		return 0;
	}

	k4a::device device = k4a::device::open(K4A_DEVICE_DEFAULT);

	k4a_device_configuration_t config = K4A_DEVICE_CONFIG_INIT_DISABLE_ALL;	//设定设备参数
	config.color_format = K4A_IMAGE_FORMAT_COLOR_BGRA32;
	config.color_resolution = K4A_COLOR_RESOLUTION_1080P;
	config.depth_mode = K4A_DEPTH_MODE_WFOV_UNBINNED;
	config.camera_fps = K4A_FRAMES_PER_SECOND_5;
	config.synchronized_images_only = true;

	device.start_cameras(&config);

	k4a::image rgbImage;
	k4a::image depthImage;
	k4a::image irImage;

	cv::Mat color_frame;
	cv::Mat depth_frame;
	cv::Mat ir_frame;

	while (1)
	{

		if (device.get_capture(&capture, std::chrono::milliseconds(0)))
		{

			rgbImage = capture.get_color_image();
			depthImage = capture.get_depth_image();
			irImage = capture.get_ir_image();

			color_frame = cv::Mat(rgbImage.get_height_pixels(), rgbImage.get_width_pixels(), CV_8UC4, rgbImage.get_buffer());
			depth_frame = cv::Mat(depthImage.get_height_pixels(), depthImage.get_width_pixels(), CV_16U, depthImage.get_buffer());
			ir_frame = cv::Mat(irImage.get_height_pixels(), irImage.get_width_pixels(), CV_16UC1, irImage.get_buffer());

			cv::imshow("color", color_frame);
			cv::imshow("depth", depth_frame);
			cv::imshow("ir", ir_frame);
	
			color_frame.release();
			depth_frame.release();
			ir_frame.release();

			capture.reset();
			if (waitKey(1) == 27)
				break;
		}
	}
	cv::destroyAllWindows();
	device.close();

	return 0;

}
